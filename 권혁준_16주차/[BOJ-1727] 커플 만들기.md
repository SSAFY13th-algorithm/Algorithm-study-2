```cpp
#include <iostream>
#include <algorithm>
using namespace std;

int N, M, A[1001]{}, B[1001]{};
int dp[1001]{}, mn[1001]{};

int main() {
	cin.tie(0)->sync_with_stdio(0);

	cin >> N >> M;
	for (int i = 1; i <= N; i++) cin >> A[i];
	for (int i = 1; i <= M; i++) cin >> B[i];
	sort(A + 1, A + N + 1);
	sort(B + 1, B + M + 1);

	if (N > M) {
		swap(N, M);
		swap(A, B);
	}

	int ans = 2e9;
	for (int j = 1; j <= M; j++) {
		dp[j] = abs(A[1] - B[j]);
		mn[j] = j == 1 ? dp[j] : min(mn[j - 1], dp[j]);
		if (N == 1) ans = min(ans, dp[j]);
	}

	for (int i = 2; i <= N; i++) {
		int ndp[1001]{}, nmn[1001]{};
		for (int j = i; j <= M; j++) {
			ndp[j] = mn[j - 1] + abs(A[i] - B[j]);
			nmn[j] = j == i ? ndp[j] : min(nmn[j - 1], ndp[j]);
			if (i == N) ans = min(ans, ndp[j]);
		}
		for (int j = 1; j <= M; j++) dp[j] = ndp[j], mn[j] = nmn[j];
	}
	cout << ans;

}
```
