```python

K, E, W, N, S = map(int, input().split())

vis = [[0 for _ in range(30)] for _ in range(30)]
cnt = 0
dp = [E,W,N,S]
dx = [1,-1,0,0]
dy = [0,0,1,-1]

def dfs(x, y, c, p):
    global vis
    global cnt
    global dp
    global dx
    global dy
    if c == K:
        cnt += p
        return
    for i in range(4):
        xx, yy = x+dx[i], y+dy[i]
        if vis[xx][yy] != 0:
            continue
        vis[xx][yy]+=1
        dfs(xx,yy,c+1,p*dp[i])
        vis[xx][yy]-=1


vis[15][15]+=1
for i in range(4):
    x, y = 15+dx[i], 15+dy[i]
    vis[x][y]+=1
    dfs(x,y,1,dp[i])
    vis[x][y]-=1

print(cnt / (100**K))

```
